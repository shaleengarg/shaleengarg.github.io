649 static inline bool nfsd4_is_solo_sequence(struct nfsd4_compoundres *resp)
650 {
651  struct nfsd4_compoundargs *args = resp->rqstp->rq_argp;
652  return resp->opcnt == 1 && args->ops[0].opnum == OP_SEQUENCE;
653 }

655 /*
656  * The session reply cache only needs to cache replies that the client
657  * actually asked us to. But it's almost free for us to cache compounds
658  * consisting of only a SEQUENCE op, so we may as well cache those too.
659  * Also, the protocol doesn't give us a convenient response in the case
660  * of a replay of a solo SEQUENCE op that wasn't cached
661  * (RETRY_UNCACHED_REP can only be returned in the second op of a
662  * compound).
663  */
664 static inline bool nfsd4_cache_this(struct nfsd4_compoundres *resp)
665 {
666  return (resp->cstate.slot->sl_flags & NFSD4_SLOT_CACHETHIS)
667  || nfsd4_is_solo_sequence(resp);
668 }

670 static inline bool nfsd4_last_compound_op(struct svc_rqst *rqstp)
671 {
672  struct nfsd4_compoundres *resp = rqstp->rq_resp;
673  struct nfsd4_compoundargs *argp = rqstp->rq_argp;
674 
675  return argp->opcnt == resp->opcnt;
