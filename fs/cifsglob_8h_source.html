762 /*
763  * When the server supports very large reads and writes via POSIX extensions,
764  * we can allow up to 2^24-1, minus the size of a READ/WRITE_AND_X header, not
765  * including the RFC1001 length.
766  *
767  * Note that this might make for "interesting" allocation problems during
768  * writeback however as we have to allocate an array of pointers for the
769  * pages. A 16M write means ~32kb page array with PAGE_SIZE == 4096.
770  *
771  * For reads, there is a similar problem as we need to allocate an array
772  * of kvecs to handle the receive, though that should only need to be done
773  * once.
774  */
775 #define CIFS_MAX_WSIZE ((1<<24) - 1 - sizeof(WRITE_REQ) + 4)
776 #define CIFS_MAX_RSIZE ((1<<24) - sizeof(READ_RSP) + 4)
777 
778 /*
779  * When the server doesn't allow large posix writes, only allow a rsize/wsize
780  * of 2^17-1 minus the size of the call header. That allows for a read or
781  * write up to the maximum size described by RFC1002.
782  */
783 #define CIFS_MAX_RFC1002_WSIZE ((1<<17) - 1 - sizeof(WRITE_REQ) + 4)
784 #define CIFS_MAX_RFC1002_RSIZE ((1<<17) - 1 - sizeof(READ_RSP) + 4)
785 
786 /*
787  * The default wsize is 1M. find_get_pages seems to return a maximum of 256
788  * pages in a single call. With PAGE_SIZE == 4k, this means we can fill
789  * a single wsize request with a single call.
790  */
791 #define CIFS_DEFAULT_IOSIZE (1024 * 1024)
792

794  * Windows only supports a max of 60kb reads and 65535 byte writes. Default to
795  * those values when posix extensions aren't in force. In actuality here, we
796  * use 65536 to allow for a write that is a multiple of 4k. Most servers seem
797  * to be ok with the extra byte even though Windows doesn't send writes that
798  * are that large.
799  *
800  * Citation:
801  *
802  * http://blogs.msdn.com/b/openspecification/archive/2009/04/10/smb-maximum-transmit-buffer-size-and-performance-tuning.aspx
803  */
804 #define CIFS_DEFAULT_NON_POSIX_RSIZE (60 * 1024)
805 #define CIFS_DEFAULT_NON_POSIX_WSIZE (65536)
