59 /* Get low bit set out of 64-bit argument, -1 if none set */
60 static inline int xfs_lowbit64(uint64_t v)
61 {
62  uint32_t w = (uint32_t)v;
63  int n = 0;
64 
65  if (w) { /* lower bits */
66  n = ffs(w);
67  } else { /* upper bits */
68  w = (uint32_t)(v >> 32);
69  if (w) {
70  n = ffs(w);
71  if (n)
72  n += 32;
73  }
74  }
75  return n - 1;
76 }
