43 static inline int gfs2_quota_lock_check(struct gfs2_inode *ip,
44  struct gfs2_alloc_parms *ap)
45 {
46  struct gfs2_sbd *sdp = GFS2_SB(&ip->i_inode);
47  int ret;
48 
49  ap->allowed = UINT_MAX; /* Assume we are permitted a whole lot */
50  if (sdp->sd_args.ar_quota == GFS2_QUOTA_OFF)
51  return 0;
52  ret = gfs2_quota_lock(ip, NO_UID_QUOTA_CHANGE, NO_GID_QUOTA_CHANGE);
53  if (ret)
54  return ret;
55  if (sdp->sd_args.ar_quota != GFS2_QUOTA_ON)
56  return 0;
57  ret = gfs2_quota_check(ip, ip->i_inode.i_uid, ip->i_inode.i_gid, ap);
58  if (ret)
59  gfs2_quota_unlock(ip);
60  return ret;
61 }
