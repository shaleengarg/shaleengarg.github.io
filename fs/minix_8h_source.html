125 static inline int minix_find_first_zero_bit(const void *vaddr, unsigned size)
126 {
127  const unsigned short *p = vaddr, *addr = vaddr;
128  unsigned short num;
129 
130  if (!size)
131  return 0;
132 
133  size >>= 4;
134  while (*p++ == 0xffff) {
135  if (--size == 0)
136  return (p - addr) << 4;
137  }
138 
139  num = *--p;
140  return ((p - addr) << 4) + ffz(num);
141 }
