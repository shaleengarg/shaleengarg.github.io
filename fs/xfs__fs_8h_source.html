38 /*
39  * Structure for XFS_IOC_GETBMAP.
40  * On input, fill in bmv_offset and bmv_length of the first structure
41  * to indicate the area of interest in the file, and bmv_entries with
42  * the number of array elements given back. The first structure is
43  * updated on return to give the offset and length for the next call.
44  */
45 #ifndef HAVE_GETBMAP
46 struct getbmap {
47  __s64 bmv_offset; /* file offset of segment in blocks */
48  __s64 bmv_block; /* starting block (64-bit daddr_t) */
49  __s64 bmv_length; /* length of segment, blocks */
50  __s32 bmv_count; /* # of entries in array incl. 1st */
51  __s32 bmv_entries; /* # of entries filled in (output) */
52 };
53 #endif

55 /*
56  * Structure for XFS_IOC_GETBMAPX. Fields bmv_offset through bmv_entries
57  * are used exactly as in the getbmap structure. The getbmapx structure
58  * has additional bmv_iflags and bmv_oflags fields. The bmv_iflags field
59  * is only used for the first structure. It contains input flags
60  * specifying XFS_IOC_GETBMAPX actions. The bmv_oflags field is filled
61  * in by the XFS_IOC_GETBMAPX command for each returned structure after
62  * the first.
63  */
64 #ifndef HAVE_GETBMAPX
65 struct getbmapx {
66  __s64 bmv_offset; /* file offset of segment in blocks */
67  __s64 bmv_block; /* starting block (64-bit daddr_t) */
68  __s64 bmv_length; /* length of segment, blocks */
69  __s32 bmv_count; /* # of entries in array incl. 1st */
70  __s32 bmv_entries; /* # of entries filled in (output). */
71  __s32 bmv_iflags; /* input flags (1st structure) */
72  __s32 bmv_oflags; /* output flags (after 1st structure)*/
73  __s32 bmv_unused1; /* future use */
74  __s32 bmv_unused2; /* future use */
75 };
76 #endif

78 /* bmv_iflags values - set by XFS_IOC_GETBMAPX caller. */

82 #define BMV_IF_DELALLOC 0x8 /* rtn status BMV_OF_DELALLOC if req */

84 #define BMV_IF_COWFORK 0x20 /* return CoW fork rather than data */

86  (BMV_IF_ATTRFORK|BMV_IF_NO_DMAPI_READ|BMV_IF_PREALLOC| \
87  BMV_IF_DELALLOC|BMV_IF_NO_HOLES|BMV_IF_COWFORK)
