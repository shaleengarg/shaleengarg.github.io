223 /* Side effect free 64 bit mod operation */
224 static inline __u32 xfs_do_mod(void *a, __u32 b, int n)
225 {
226  switch (n) {
227  case 4:
228  return *(__u32 *)a % b;
229  case 8:
230  {
231  __u64 c = *(__u64 *)a;
232  return do_div(c, b);
233  }
234  }
235 
236  /* NOTREACHED */
237  return 0;
238 }
