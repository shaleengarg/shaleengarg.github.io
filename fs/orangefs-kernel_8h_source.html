75 /*
76  * valid orangefs kernel operation states
77  *
78  * unknown - op was just initialized
79  * waiting - op is on request_list (upward bound)
80  * inprogr - op is in progress (waiting for downcall)
81  * serviced - op has matching downcall; ok
82  * purged - op has to start a timer since client-core
83  * exited uncleanly before servicing op
84  * given up - submitter has given up waiting for it
85  */
86 enum orangefs_vfs_op_states {
87  OP_VFS_STATE_UNKNOWN = 0,
88  OP_VFS_STATE_WAITING = 1,
89  OP_VFS_STATE_INPROGR = 2,
90  OP_VFS_STATE_SERVICED = 4,
91  OP_VFS_STATE_PURGED = 8,
92  OP_VFS_STATE_GIVEN_UP = 16,
93 };

167 static inline void set_op_state_purged(struct orangefs_kernel_op_s *op)
168 {
169  spin_lock(&op->lock);
170  if (unlikely(op_is_cancel(op))) {
171  list_del_init(&op->list);
172  spin_unlock(&op->lock);
173  put_cancel(op);
174  } else {
175  op->op_state |= OP_VFS_STATE_PURGED;
176  complete(&op->waitq);
177  spin_unlock(&op->lock);
178  }

237 /* ino_t descends from "unsigned long", 8 bytes, 64 bits. */
238 static inline ino_t orangefs_khandle_to_ino(struct orangefs_khandle *khandle)
239 {
240  union {
241  unsigned char u[8];
242  __u64 ino;
243  } ihandle;
244 
245  ihandle.u[0] = khandle->u[0] ^ khandle->u[4];
246  ihandle.u[1] = khandle->u[1] ^ khandle->u[5];
247  ihandle.u[2] = khandle->u[2] ^ khandle->u[6];
248  ihandle.u[3] = khandle->u[3] ^ khandle->u[7];
249  ihandle.u[4] = khandle->u[12] ^ khandle->u[8];
250  ihandle.u[5] = khandle->u[13] ^ khandle->u[9];
251  ihandle.u[6] = khandle->u[14] ^ khandle->u[10];
252  ihandle.u[7] = khandle->u[15] ^ khandle->u[11];
253 
254  return ihandle.ino;
255 }

262 static inline int is_root_handle(struct inode *inode)
263 {
264  gossip_debug(GOSSIP_DCACHE_DEBUG,
265  "%s: root handle: %pU, this handle: %pU:\n",
266  __func__,
267  &ORANGEFS_SB(inode->i_sb)->root_khandle,
268  get_khandle_from_ino(inode));
269 
270  if (ORANGEFS_khandle_cmp(&(ORANGEFS_SB(inode->i_sb)->root_khandle),
271  get_khandle_from_ino(inode)))
272  return 0;
273  else
274  return 1;
275 }
