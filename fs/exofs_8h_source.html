209 /* exofs_init_comps will initialize an ore_components device array
210  * pointing to a single ore_comp struct, and a round-robin view
211  * of the device table.
212  * The first device of each inode is the [inode->ino % num_devices]
213  * and the rest of the devices sequentially following where the
214  * first device is after the last device.
215  * It is assumed that the global device array at @sbi is twice
216  * bigger and that the device table repeats twice.
217  * See: exofs_read_lookup_dev_table()
218  */
219 static inline void exofs_init_comps(struct ore_components *oc,
220  struct ore_comp *one_comp,
221  struct exofs_sb_info *sbi, osd_id oid)
222 {
223  unsigned dev_mod = (unsigned)oid, first_dev;
224 
225  one_comp->obj.partition = sbi->one_comp.obj.partition;
226  one_comp->obj.id = oid;
227  exofs_make_credential(one_comp->cred, &one_comp->obj);
228 
229  oc->first_dev = 0;
230  oc->numdevs = sbi->layout.group_width * sbi->layout.mirrors_p1 *
231  sbi->layout.group_count;
232  oc->single_comp = EC_SINGLE_COMP;
233  oc->comps = one_comp;
234 
235  /* Round robin device view of the table */
236  first_dev = (dev_mod * sbi->layout.mirrors_p1) % sbi->oc.numdevs;
237  oc->ods = &sbi->oc.ods[first_dev];
238 }
