133 /*
134  * Locate the struct page for both the matching buddy in our
135  * pair (buddy1) and the combined O(n+1) page they form (page).
136  *
137  * 1) Any buddy B1 will have an order O twin B2 which satisfies
138  * the following equation:
139  * B2 = B1 ^ (1 << O)
140  * For example, if the starting buddy (buddy2) is #8 its order
141  * 1 buddy is #10:
142  * B2 = 8 ^ (1 << 1) = 8 ^ 2 = 10
143  *
144  * 2) Any buddy B will have an order O+1 parent P which
145  * satisfies the following equation:
146  * P = B & ~(1 << O)
147  *
148  * Assumption: *_mem_map is contiguous at least up to MAX_ORDER
149  */
150 static inline unsigned long
151 __find_buddy_pfn(unsigned long page_pfn, unsigned int order)
152 {
153  return page_pfn ^ (1 << order);
154 }

225 /*
226  * This function returns the order of a free page in the buddy system. In
227  * general, page_zone(page)->lock must be held by the caller to prevent the
228  * page from being allocated in parallel and returning garbage as the order.
229  * If a caller does not hold page_zone(page)->lock, it must guarantee that the
230  * page cannot be allocated or merged in parallel. Alternatively, it must
231  * handle invalid values gracefully, and use page_order_unsafe() below.
232  */
233 static inline unsigned int page_order(struct page *page)
234 {
235  /* PageBuddy() must be checked by the caller */
236  return page_private(page);
237 }

239 /*
240  * Like page_order(), but for callers who cannot afford to hold the zone lock.
241  * PageBuddy() should be checked first by the caller to minimize race window,
242  * and invalid values must be handled gracefully.
243  *
244  * READ_ONCE is used so that if the caller assigns the result into a local
245  * variable and e.g. tests it for valid range before using, the compiler cannot
246  * decide to remove the variable and inline the page_private(page) multiple
247  * times, potentially observing different values in the tests and the actual
248  * use of the result.
249  */
250 #define page_order_unsafe(page) READ_ONCE(page_private(page))

369 /*
370  * Return the mem_map entry representing the 'offset' subpage within
371  * the maximally aligned gigantic page 'base'. Handle any discontiguity
372  * in the mem_map at MAX_ORDER_NR_PAGES boundaries.
373  */
374 static inline struct page *mem_map_offset(struct page *base, int offset)
375 {
376  if (unlikely(offset >= MAX_ORDER_NR_PAGES))
377  return nth_page(base, offset);
378  return base + offset;
379 }
380 
381 /*
382  * Iterator over all subpages within the maximally aligned gigantic
383  * page 'base'. Handle any discontiguity in the mem_map.
384  */
385 static inline struct page *mem_map_next(struct page *iter,
386  struct page *base, int offset)
387 {
388  if (unlikely((offset & (MAX_ORDER_NR_PAGES - 1)) == 0)) {
389  unsigned long pfn = page_to_pfn(base) + offset;
390  if (!pfn_valid(pfn))
391  return NULL;
392  return pfn_to_page(pfn);
393  }
394  return iter + 1;
395 }
