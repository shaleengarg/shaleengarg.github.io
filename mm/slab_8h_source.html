257  rcu_read_lock();

359 static inline struct kmem_cache *cache_from_obj(struct kmem_cache *s, void *x)
360 {
361  struct kmem_cache *cachep;
362  struct page *page;
363 
364  /*
365  * When kmemcg is not being used, both assignments should return the
366  * same value. but we don't want to pay the assignment price in that
367  * case. If it is not compiled in, the compiler should be smart enough
368  * to not do even the assignment. In that case, slab_equal_or_root
369  * will also be a constant.
370  */
371  if (!memcg_kmem_enabled() &&
372  !unlikely(s->flags & SLAB_CONSISTENCY_CHECKS))
373  return s;
374 
375  page = virt_to_head_page(x);
376  cachep = page->slab_cache;
377  if (slab_equal_or_root(cachep, s))
378  return cachep;
379 
380  pr_err("%s: Wrong slab cache. %s but object is from %s\n",
381  __func__, s->name, cachep->name);
382  WARN_ON_ONCE(1);
383  return s;
384 }

386 static inline size_t slab_ksize(const struct kmem_cache *s)
387 {
388 #ifndef CONFIG_SLUB
389  return s->object_size;
390 
391 #else /* CONFIG_SLUB */
392 # ifdef CONFIG_SLUB_DEBUG
393  /*
394  * Debugging requires use of the padding between object
395  * and whatever may come after it.
396  */
397  if (s->flags & (SLAB_RED_ZONE | SLAB_POISON))
398  return s->object_size;
399 # endif
400  if (s->flags & SLAB_KASAN)
401  return s->object_size;
402  /*
403  * If we have the need to store the freelist pointer
404  * back there or track user information then we can
405  * only use the space before that information.
406  */
407  if (s->flags & (SLAB_TYPESAFE_BY_RCU | SLAB_STORE_USER))
408  return s->inuse;
409  /*
410  * Else we can use all the padding etc for the allocation
411  */
412  return s->size;
413 #endif
414 }

416 static inline struct kmem_cache *slab_pre_alloc_hook(struct kmem_cache *s,
417  gfp_t flags)
418 {
419  flags &= gfp_allowed_mask;
420 
421  fs_reclaim_acquire(flags);
422  fs_reclaim_release(flags);
423 
424  might_sleep_if(gfpflags_allow_blocking(flags));
425 
426  if (should_failslab(s, flags))
427  return NULL;
428 
429  if (memcg_kmem_enabled() &&
430  ((flags & __GFP_ACCOUNT) || (s->flags & SLAB_ACCOUNT)))
431  return memcg_kmem_get_cache(s);
432 
433  return s;
434 }

436 static inline void slab_post_alloc_hook(struct kmem_cache *s, gfp_t flags,
437  size_t size, void **p)
438 {
439  size_t i;
440 
441  flags &= gfp_allowed_mask;
442  for (i = 0; i < size; i++) {
443  void *object = p[i];
444 
445  kmemleak_alloc_recursive(object, s->object_size, 1,
446  s->flags, flags);
447  kasan_slab_alloc(s, object, flags);
448  }
449 
450  if (memcg_kmem_enabled())
451  memcg_kmem_put_cache(s);
452 }
